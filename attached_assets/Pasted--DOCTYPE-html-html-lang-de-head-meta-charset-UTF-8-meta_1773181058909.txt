    function buildSteeringBars(level) {
      return Array.from({ length: 6 }, (_, index) => {
        const isFilled = index < level;
        return `<span class="bar ${isFilled ? "filled" : ""}"></span>`;
      }).join("");
    }

    }

    function renderTeamCheckPreview(roleType, variantKey) {
      const roleData = TEAMCHECK_PREVIEW_DATA[roleType];
      if (!roleData) {
        console.error("Unbekannter roleType:", roleType);
        return;
      }

      const variant = roleData[variantKey];
      if (!variant) {
        console.error("Unbekannte Variante:", variantKey);
        return;
      }

      const roleChip = document.getElementById("teamcheckRoleChip");
      const resultBadge = document.getElementById("teamcheckResultBadge");
      const indicatorText = document.getElementById("teamcheckIndicatorText");
      const steeringLevel = document.getElementById("teamcheckIntensityLevel");
      const steeringText = document.getElementById("teamcheckIntensityText");
      const steeringBars = document.getElementById("teamcheckIntensityBars");
      const steeringDescription = document.getElementById("teamcheckIntensityDescription");

      roleChip.textContent = getRoleLabel(roleType);
      resultBadge.textContent = variant.result;
      resultBadge.className = `teamcheck-preview-result ${variant.resultClass}`;

      indicatorText.textContent = variant.indicatorText;
      steeringLevel.textContent = `${variant.steeringLevel} von 6`;
      steeringText.textContent = variant.steeringLabel;
      steeringBars.innerHTML = buildSteeringBars(variant.steeringLevel);
      steeringBars.setAttribute(
        "aria-label",
        `Steuerungsintensität ${variant.steeringLevel} von 6`
      );
      steeringDescription.textContent = variant.steeringDescription;
    }

    function getResultCategoryFromVariant(variantKey) {
      const numericVariant = Number(variantKey);

      if (numericVariant >= 1 && numericVariant <= 3) {
        return "geeignet";
      }

      if (numericVariant >= 4 && numericVariant <= 8) {
        return "bedingt geeignet";
      }

      return "nicht geeignet";
    }
