// Im Fließtext wird verständlich und natürlich formuliert.

export type ComponentKey = "IMP" | "INT" | "ANA";

export type ResultKey =
  | "IMP_INT_ANA"
  | "IMP_ANA_INT"
  | "INT_IMP_ANA"
  | "INT_ANA_IMP"
  | "ANA_IMP_INT"
  | "ANA_INT_IMP"
  | "IMP_INT__ANA"
  | "IMP_ANA__INT"
  | "INT_IMP__ANA"
  | "INT_ANA__IMP"
  | "ANA_IMP__INT"
  | "ANA_INT__IMP"
  | "BALANCED";

export type DominanceType =
  | "single"
  | "top_dual"
  | "bottom_dual"
  | "balanced";

export type RoleProfileInput = {
  impulsiv: number;
  intuitiv: number;
  analytisch: number;
  isLeadershipRole?: boolean;
};

export type RankedItem = {
  key: ComponentKey;
  value: number;
};

export type ComputedRoleProfile = {
  resultKey: ResultKey;
  dominanceType: DominanceType;
  sorted: RankedItem[];
  topDiff: number;
  bottomDiff: number;
};

export type ResultText = {
  headline: string;
  body: string[];
  leadership?: string;
};

const DUAL_THRESHOLD = 5;
const BALANCED_THRESHOLD = 5;

function getHeadlineLabel(key: ComponentKey): string {
  switch (key) {
    case "IMP":
      return "Impulsiv";
    case "INT":
      return "Intuitiv";
    case "ANA":
      return "Analytisch";
  }
}

function getSortedProfile(input: RoleProfileInput): RankedItem[] {
  return [
    { key: "IMP", value: input.impulsiv },
    { key: "INT", value: input.intuitiv },
    { key: "ANA", value: input.analytisch },
  ].sort((a, b) => b.value - a.value);
}

export function computeRoleProfile(
  input: RoleProfileInput
): ComputedRoleProfile {
  const sorted = getSortedProfile(input);

  const topDiff = sorted[0].value - sorted[1].value;
  const bottomDiff = sorted[1].value - sorted[2].value;
  const spread = sorted[0].value - sorted[2].value;

  if (spread <= BALANCED_THRESHOLD) {
    return {
      resultKey: "BALANCED",
      dominanceType: "balanced",
      sorted,
      topDiff,
      bottomDiff,
    };
  }

  // Top-Doppeldominanz: Platz 1 und 2 liegen max. 5 Punkte auseinander
  // Dann prägen beide den Alltag, Platz 3 kommt eher unter Druck ins Spiel.
  if (topDiff <= DUAL_THRESHOLD) {
    const resultKey = `${sorted[0].key}_${sorted[1].key}__${sorted[2].key}` as ResultKey;
    return {
      resultKey,
      dominanceType: "top_dual",
      sorted,
      topDiff,
      bottomDiff,
    };
  }

  // Bottom-Doppeldominanz: Platz 2 und 3 liegen max. 5 Punkte auseinander
  // Dann prägt Platz 1 den Alltag, unter Druck können Platz 2 und 3 konkurrieren.
  if (bottomDiff <= DUAL_THRESHOLD) {
    const resultKey = `${sorted[0].key}_${sorted[1].key}__${sorted[2].key}` as ResultKey;
    return {
      resultKey,
      dominanceType: "bottom_dual",
      sorted,
      topDiff,
      bottomDiff,
    };
  }

  // Klare Rangfolge
  const resultKey = `${sorted[0].key}_${sorted[1].key}_${sorted[2].key}` as ResultKey;
  return {
    resultKey,
    dominanceType: "single",
    sorted,
    topDiff,
    bottomDiff,
  };
}

export function getRoleResultText(input: RoleProfileInput): ResultText {
  const computed = computeRoleProfile(input);
  const baseText = roleResultTexts[computed.resultKey];

  if (!input.isLeadershipRole) {
    return {
      headline: baseText.headline,
      body: baseText.body,
    };
  }

  return baseText;
}

export function renderRoleResultHtml(input: RoleProfileInput): string {
  const text = getRoleResultText(input);

  const paragraphs = [...text.body];
  if (input.isLeadershipRole && text.leadership) {
    paragraphs.push(text.leadership);
  }

  return `
    <section class="jobcheck-result">
      <h3 class="jobcheck-result__headline">${text.headline}</h3>
      ${paragraphs
        .map((p) => `<p class="jobcheck-result__text">${p}</p>`)
        .join("\n")}
    </section>
  `.trim();
}

export function renderRoleResultObject(input: RoleProfileInput) {
  const computed = computeRoleProfile(input);
  const text = getRoleResultText(input);

  return {
    resultKey: computed.resultKey,
    dominanceType: computed.dominanceType,
    headline: text.headline,
    paragraphs: input.isLeadershipRole && text.leadership
      ? [...text.body, text.leadership]
      : [...text.body],
    sorted: computed.sorted,
    topDiff: computed.topDiff,
    bottomDiff: computed.bottomDiff,
  };
}
