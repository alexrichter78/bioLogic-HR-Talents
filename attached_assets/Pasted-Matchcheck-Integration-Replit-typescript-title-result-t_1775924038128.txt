export function getImpactCards(result: UnifiedMatchcheckResult) {
  return result.texts.impactAreas.map((area) => ({
    key: area.key,
    title: area.title,
    status: area.label,
    severity: area.severity,
    roleNeed: area.roleNeed,
    candidatePattern: area.candidatePattern,
    interpretation: area.interpretation,
  }));
}

Was die neue Integrationsdatei macht

Sie sorgt dafür, dass du in Replit nur noch eine zentrale Funktion aufrufst:

computeUnifiedMatchcheck(...)

Diese Funktion:

ruft zuerst die Fit-Berechnung auf
leitet daraus automatisch den FitSubtype ab
gibt die Daten an die Text-Engine
liefert dir ein einheitliches Ergebnisobjekt für die UI

eine Textlogik
